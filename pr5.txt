radius = float(input("input radius : "))

pi = 3.14159

print("the area of the circle with radius: " + str(radius) + " is= " + str(pi * radius**2))
print("the diameter of the circle with radius: " + str(radius) + " is= " + str(2 * radius))
print("the circumference of the circle with radius: " + str(radius) + " is= " + str(2 * pi * radius))
